char *
strncpy(char *dst, const char *src, size_t n)
{
    if (n != 0) {
        char *d = dst;
        const char *s = src;

        do {
            if ((*d++ = *s++) == 0) {
                while (--n != 0) *d++ = 0;
                break;
            }
        } while (--n != 0);
    }
    return (dst);
}

https://github.com/openbsd/src/blob/master/lib/libc/string/strlcpy.c 


size_t
strlcpy(char *dst, const char *src, size_t n)
{
    const char *osrc = src;
    size_t nleft = n;

    if (nleft != 0) {
        while (--nleft != 0) {
            if ((*dst++ = *src++) == '\0')
                break;
        }
    }

    if (nleft == 0) {
        if (n != 0)
            *dst = '\0';
        while (*src++);
    }

    return(src - osrc - 1);
}

strncpy VS strlcpy

  1. both strncpy and strlcpy behave as strcpy if n is no less than src's size(for src, size means its strlen plus 1)
  2. if n is zero, dst is not affected
  4. strlcpy returns size_t but some implemention may return number of bytes actually copied instead; strncpy always return dst
  5. strlcpy will set dst[n-1] as \0 if n is less than src's size

code conversion case 00 

strncpy(dest, src, n);
dest[n - 1] = '\0';

==>

strlcpy(dest, src, n);

code conversion case 01 

strncpy(dest, src, n - 1);
dest[n - 1] = '\0';

==>

strlcpy(dest, src, n);
// even if there is no  `dest[n - 1] = '\0';` line, it's still possiblely we shall strlcpy with n instead of 'n-1', but making sure n is no larger than dst's capacity
